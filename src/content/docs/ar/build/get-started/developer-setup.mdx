import { CardGrid, LinkCard, Steps, TabItem, Tabs } from '@astrojs/starlight/components';

إليك طريقة سهلة لإعداد بيئتك اعتماداً على نوع التطوير.

<Tabs>
  {/* Frontend */}

  <TabItem label="الواجهة الأمامية">
    <Steps>
      1. تهيئة مشروع الواجهة الأمامية

         إليك بعض الأمثلة على الخيارات الشائعة:

         <Tabs>
           <TabItem label="Next.js">
             ```shellscript filename="Terminal"
             pnpx create-next-app@latest
             ```
           </TabItem>

           <TabItem label="Vite (TS)">
             ```shellscript filename="Terminal"
             pnpx create vite my-aptos-app --template react-ts
             ```
           </TabItem>
         </Tabs>

      2. تثبيت @aptos-labs/ts-sdk

         ```shellscript npm2yarn
         npm i @aptos-labs/ts-sdk
         ```

      3. إعداد TS SDK

         <CardGrid>
           <LinkCard
             href="/build/sdks/ts-sdk/quickstart"
             title="بدء سريع لـ TS SDK"
             description="تعرف على كيفية إعداد حسابك، الشبكة، استخدام
         الصنبور، إرسال / محاكاة المعاملات، والمزيد"
           />
         </CardGrid>

      4. أنشئ تطبيقك!

         إعداد المطور لاستخدام Aptos في واجهتك الأمامية مكتمل الآن.
         تحقق من أدواتنا الأخرى التي تبسط عملية التطوير

         <CardGrid>
           <LinkCard
             href="/build/indexer"
             title="الفهرس"
             description="استعلام حالة السلسلة بكفاءة مثل الأرصدة،
         نشاط المعاملات، بيانات الرموز، والمزيد"
           />

           <LinkCard
             href="https://github.com/aptos-labs/aptos-ts-sdk/tree/main/examples/typescript"
             title="أمثلة TS SDK"
             description="20+ مثالاً على كيفية استخدام
         TS SDK"
             target="_blank"
           />

           <LinkCard href="https://geomi.dev/" title="Geomi" description="هل وصلت إلى حدود المعدل لـ Fullnode API / الفهرسات؟ احصل على مفتاح API هنا" target="_blank" />
         </CardGrid>
    </Steps>
  </TabItem>

  {/* Smart Contract */}

  <TabItem label="العقد الذكي">
    <Steps>
      1. تثبيت CLI

         <CardGrid>
           <LinkCard href="/build/cli" title="Aptos CLI" description="تعليمات تثبيت Aptos CLI" />
         </CardGrid>

      2. إعداد المحرر أو IDE

         أضف الامتدادات التالية إلى محررك المختار لتسهيل تطوير Move

         <Tabs>
           <TabItem label="IDEs JetBrains">
             <CardGrid>
               <LinkCard
                 href="https://plugins.jetbrains.com/plugin/14721-move-language"
                 title="Move on Aptos"
                 description="خادم اللغة ومظلل الصيغة لـ
         JetBrains IDEs مثل CLion، Rust Rover،
         WebStorm، IntelliJ"
                 target="_blank"
               />
             </CardGrid>
           </TabItem>

           <TabItem label="VSCode">
             <CardGrid>
               <LinkCard href="https://marketplace.visualstudio.com/items?itemName=AptosLabs.move-on-aptos" title="امتداد Move on Aptos" description="IDE Aptos Move كامل لـ VSCode" target="_blank" />
             </CardGrid>
           </TabItem>

           <TabItem label="Cursor، Kiro، إلخ">
             <CardGrid>
               <LinkCard href="https://open-vsx.org/extension/aptoslabs/move-on-aptos" title="امتداد Move on Aptos" description="IDE Aptos Move كامل لمنصات OpenVSX" target="_blank" />
             </CardGrid>
           </TabItem>
         </Tabs>

      3. إنشاء عقد ذكي

         انتقل إلى مجلد التطبيق الخاص بك وقم بتهيئة عقد ذكي جديد بإجراء:

         ```shellscript filename="Terminal"
         aptos move init --name my_todo_list
         ```

      4. بناء، تجميع ونشر العقد الذكي!

         إعداد المطور لاستخدام Aptos للعقود الذكية مكتمل الآن.
         للمزيد من المعلومات، راجع رابط البرنامج التعليمي لـ Dapp أدناه

         <CardGrid>
           <LinkCard
             href="/build/guides/build-e2e-dapp/1-create-smart-contract#what-is-a-movetoml-file"
             title="دليل إنشاء العقد الذكي"
             description="دليل قائمة مهام سهل لإعداد عقد
         ذكي باستخدام Move"
           />
         </CardGrid>
    </Steps>
  </TabItem>

  {/* create-aptos-dapp */}

  <TabItem label="إنشاء DApp Aptos">
    <Steps>
      1. تثبيت create-aptos-dapp

         لتثبيت dApp من قالب في ثوانٍ، قم بتنفيذ الأمر أدناه:

         <Tabs>
           { /* npx */ }

           <TabItem label="npx">
             ```shellscript filename="Terminal"
             npx create-aptos-dapp@latest
             ```
           </TabItem>

           { /* pnpx */ }

           <TabItem label="pnpx">
             ```shellscript filename="Terminal"
             pnpx create-aptos-dapp@latest
             ```
           </TabItem>
         </Tabs>

      2. اتبع التعليمات

         اتبع تعليمات CLI لتحديد اسم، [قالب](/build/create-aptos-dapp#templates)، وشبكة لـ dApp الجديدة الخاصة بك.

         ![cad](~/images/cad-video.gif)

      3. ابدأ في بناء وتخصيص dApp الجديدة الخاصة بك!

         انتقل إلى مشروعك الجديد وافتحه في IDE المفضل لديك للمتابعة في البناء!

         اتبع ملف `README.md` المُنشأ للخطوات التالية.

      4. استمر في القراءة

         <CardGrid>
           <LinkCard href="/build/create-aptos-dapp" title="إنشاء DApp Aptos" description="احصل على مزيد من المعلومات حول الأداة" />

           <LinkCard href="/build/create-aptos-dapp#templates" title="القوالب" description="تصفح القوالب المعدة مسبقاً" />

           <LinkCard href="/build/create-aptos-dapp/faq" title="الأسئلة الشائعة" description="احصل على مساعدة للمشكلات والأسئلة الشائعة" />
         </CardGrid>
    </Steps>
  </TabItem>
</Tabs>
