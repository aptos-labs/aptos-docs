المعاملات هي الطريقة الأساسية لتغيير البيانات على بلوك تشين أبتوس. فكر فيها مثل إرسال طرد: تحتاج إلى تحديد ما ترسله، إلى من يذهب، ثم تتبعه حتى يتم تأكيد التسليم. في مصطلحات البلوك تشين، تسمح لك المعاملات بنقل العملات، واستدعاء دوال العقود الذكية، وتحديث الحالة على السلسلة.

سيرشدك هذا البرنامج التعليمي خلال إنشاء وإرسال معاملتك الأولى على بلوك تشين أبتوس. ستتعلم كيفية:

1. إعداد بيئة التطوير الخاصة بك
2. إنشاء حسابات تجريبية وتمويلها
3. بناء معاملة لنقل العملات
4. محاكاة المعاملة لتقدير التكاليف
5. توقيع وإرسال المعاملة
6. التحقق من تنفيذ المعاملة بنجاح

<Aside type="note">
  يبني هذا البرنامج التعليمي على مفاهيم من بلوك تشين أبتوس. إذا كنت جديداً على تطوير البلوك تشين، لا تقلق - سنشرح المفاهيم الرئيسية على طول الطريق.

  يمكنك الانتقال إلى نموذج الكود الكامل [هنا](#نموذج-الكود-الكامل) أو متابعة القراءة للحصول على شرح خطوة بخطوة.
</Aside>

## 6. إنشاء وإرسال المعاملة

الآن لننشئ معاملة لنقل APT من Alice إلى Bob.

<Tabs>
  <TabItem label="TypeScript">
    ```typescript filename="transaction.ts"
    async function transferCoins(alice: Account, bob: Account) {
      const transaction = await aptos.transaction.build.simple({
        sender: alice.accountAddress,
        data: {
          function: "0x1::coin::transfer",
          typeArguments: ["0x1::aptos_coin::AptosCoin"],
          functionArguments: [bob.accountAddress, 1_000_000], // 0.01 APT
        },
      });
      
      // محاكاة المعاملة للتحقق من رسوم الوقود
      const simulation = await aptos.transaction.simulate.simple({
        signerPublicKey: alice.publicKey,
        transaction,
      });
      
      console.log("نتيجة المحاكاة:", simulation);
      
      // توقيع وإرسال المعاملة
      const committedTransaction = await aptos.signAndSubmitTransaction({
        signer: alice,
        transaction,
      });
      
      console.log("تم إرسال المعاملة:", committedTransaction.hash);
      
      // انتظار اكتمال المعاملة
      await aptos.waitForTransaction({ transactionHash: committedTransaction.hash });
      
      console.log("تمت المعاملة!");
      
      return committedTransaction.hash;
    }
    ```
  </TabItem>

  <TabItem label="Python">
    ```python filename="transaction.py"
    async def transfer_coins(alice, bob):
        # بناء المعاملة
        payload = EntryFunction.natural(
            "0x1::coin",
            "transfer",
            [TypeTag(StructTag.from_str("0x1::aptos_coin::AptosCoin"))],
            [
                TransactionArgument(bob.address(), "address"),
                TransactionArgument(1_000_000, "u64"),  # 0.01 APT
            ],
        )
        
        # إنشاء وإرسال المعاملة
        signed_transaction = await client.create_bcs_signed_transaction(
            alice, TransactionPayload(payload)
        )
        
        result = await client.submit_bcs_transaction(signed_transaction)
        
        print(f"تم إرسال المعاملة: {result}")
        
        # انتظار اكتمال المعاملة
        await client.wait_for_transaction(result)
        
        print("تمت المعاملة!")
        
        return result
    ```
  </TabItem>
</Tabs>

## نموذج الكود الكامل

<Tabs>
  <TabItem label="TypeScript">
    ```typescript filename="transaction.ts"
    import {
      Account,
      Aptos,
      AptosConfig,
      Network,
    } from "@aptos-labs/ts-sdk";

    const config = new AptosConfig({ network: Network.TESTNET });
    const aptos = new Aptos(config);

    async function main() {
      console.log("بدء عرض المعاملة الأولى...");
      
      // 1. إنشاء حسابات
      const alice = Account.generate();
      const bob = Account.generate();
      
      console.log("عنوان Alice:", alice.accountAddress.toString());
      console.log("عنوان Bob:", bob.accountAddress.toString());
      
      // 2. تمويل Alice
      await aptos.fundAccount({
        accountAddress: alice.accountAddress,
        amount: 100_000_000, // 1 APT
      });
      
      console.log("تم تمويل Alice");
      
      // 3. التحقق من الأرصدة الأولية
      console.log("\n--- الأرصدة الأولية ---");
      await checkBalances(alice, bob);
      
      // 4. نقل العملات
      console.log("\n--- نقل العملات ---");
      const txnHash = await transferCoins(alice, bob);
      
      // 5. التحقق من الأرصدة النهائية
      console.log("\n--- الأرصدة النهائية ---");
      await checkBalances(alice, bob);
      
      console.log(`\n✅ نجحت المعاملة! المعرف: ${txnHash}`);
    }

    async function transferCoins(alice: Account, bob: Account) {
      const transaction = await aptos.transaction.build.simple({
        sender: alice.accountAddress,
        data: {
          function: "0x1::coin::transfer",
          typeArguments: ["0x1::aptos_coin::AptosCoin"],
          functionArguments: [bob.accountAddress, 1_000_000], // 0.01 APT
        },
      });
      
      const committedTransaction = await aptos.signAndSubmitTransaction({
        signer: alice,
        transaction,
      });
      
      await aptos.waitForTransaction({ transactionHash: committedTransaction.hash });
      
      return committedTransaction.hash;
    }

    async function checkBalances(alice: Account, bob: Account) {
      const aliceBalance = await aptos.getAccountAPTAmount({
        accountAddress: alice.accountAddress,
      });
      
      const bobBalance = await aptos.getAccountAPTAmount({
        accountAddress: bob.accountAddress,
      });
      
      console.log(`رصيد Alice: ${aliceBalance} Octas`);
      console.log(`رصيد Bob: ${bobBalance} Octas`);
    }

    // تشغيل الدالة الرئيسية
    main().catch(console.error);
    ```
  </TabItem>

  <TabItem label="Python">
    ```python filename="transaction.py"
    import asyncio
    from aptos_sdk.account import Account
    from aptos_sdk.async_client import RestClient

    NODE_URL = "https://fullnode.testnet.aptoslabs.com/v1"

    async def main():
        print("بدء عرض المعاملة الأولى...")
        
        client = RestClient(NODE_URL)
        
        # 1. إنشاء حسابات
        alice = Account.generate()
        bob = Account.generate()
        
        print(f"عنوان Alice: {alice.address()}")
        print(f"عنوان Bob: {bob.address()}")
        
        # 2. تمويل Alice
        await client.fund_account(alice.address(), 100_000_000)  # 1 APT
        print("تم تمويل Alice")
        
        # 3. التحقق من الأرصدة الأولية
        print("\n--- الأرصدة الأولية ---")
        await check_balances(client, alice, bob)
        
        # 4. نقل العملات
        print("\n--- نقل العملات ---")
        txn_hash = await transfer_coins(client, alice, bob)
        
        # 5. التحقق من الأرصدة النهائية
        print("\n--- الأرصدة النهائية ---")
        await check_balances(client, alice, bob)
        
        print(f"\n✅ نجحت المعاملة! المعرف: {txn_hash}")

    async def check_balances(client, alice, bob):
        alice_balance = await client.account_balance(alice.address())
        bob_balance = await client.account_balance(bob.address())
        
        print(f"رصيد Alice: {alice_balance} Octas")
        print(f"رصيد Bob: {bob_balance} Octas")

    # تشغيل الدالة الرئيسية
    if __name__ == "__main__":
        asyncio.run(main())
    ```
  </TabItem>
</Tabs>
