import { CardGrid, LinkCard, Steps, TabItem, Tabs } from '@astrojs/starlight/components';

Hier ist eine einfache Möglichkeit, Ihre Umgebung je nach Art der Entwicklung einzurichten.

<Tabs>
  {/* Frontend */}

  <TabItem label="Frontend">
    <Steps>
      1. Frontend-Projekt initialisieren

         Hier sind einige Beispiele beliebter Optionen:

         <Tabs>
           <TabItem label="Next.js">
             ```shellscript filename="Terminal"
             pnpx create-next-app@latest
             ```
           </TabItem>

           <TabItem label="Vite (TS)">
             ```shellscript filename="Terminal"
             pnpx create vite my-aptos-app --template react-ts
             ```
           </TabItem>
         </Tabs>

      2. @aptos-labs/ts-sdk installieren

         ```shellscript npm2yarn
         npm i @aptos-labs/ts-sdk
         ```

      3. TS SDK einrichten

         <CardGrid>
           <LinkCard href="/de/build/sdks/ts-sdk/quickstart" title="TS SDK Schnellstart" description="Erfahren Sie, wie Sie Ihr Konto einrichten, das Netzwerk verwenden, den Faucet nutzen, Transaktionen senden/simulieren und mehr" />
         </CardGrid>

      4. Entwickeln Sie Ihre App!

         Das Entwickler-Setup für die Verwendung von Aptos in Ihrem Frontend ist nun abgeschlossen.
         Schauen Sie sich unsere anderen Tools an, die den Entwicklungsprozess optimieren

         <CardGrid>
           <LinkCard href="/de/build/indexer" title="Indexer" description="Effizient On-Chain-Status abfragen wie Guthaben, Transaktionsaktivität, Token-Daten und mehr" />

           <LinkCard href="https://github.com/aptos-labs/aptos-ts-sdk/tree/main/examples/typescript" title="TS SDK Beispiele" description="20+ Beispiele zur Verwendung des TS SDK" target="_blank" />

           <LinkCard href="https://build.aptoslabs.com/" title="Aptos Build" description="Stoßen Sie an Rate Limits für Fullnode API / Indexer? Holen Sie sich hier einen API-Schlüssel" target="_blank" />
         </CardGrid>
    </Steps>
  </TabItem>

  {/* Smart Contract */}

  <TabItem label="Smart Contract">
    <Steps>
      1. CLI installieren

         <CardGrid>
           <LinkCard href="/de/build/cli" title="Aptos CLI" description="Anweisungen zur Installation der Aptos CLI" />
         </CardGrid>

      2. Editor oder IDE einrichten

         Fügen Sie die folgenden Erweiterungen zu Ihrem bevorzugten Editor hinzu, um die Move-Entwicklung zu erleichtern

         <Tabs>
           <TabItem label="JetBrains IDEs">
             <CardGrid>
               <LinkCard href="https://plugins.jetbrains.com/plugin/14721-move-language" title="Move on Aptos" description="Sprachserver und Syntax-Highlighter für JetBrains IDEs wie CLion, Rust Rover, WebStorm, IntelliJ" target="_blank" />
             </CardGrid>
           </TabItem>

           <TabItem label="VSCode">
             <CardGrid>
               <LinkCard href="https://marketplace.visualstudio.com/items?itemName=AptosLabs.move-on-aptos" title="Move on Aptos Erweiterung" description="Vollständige Aptos Move IDE für VSCode" target="_blank" />
             </CardGrid>
           </TabItem>

           <TabItem label="Cursor, Kiro, etc.">
             <CardGrid>
               <LinkCard href="https://open-vsx.org/extension/aptoslabs/move-on-aptos" title="Move on Aptos Erweiterung" description="Vollständige Aptos Move IDE für OpenVSX-basierte Plattformen" target="_blank" />
             </CardGrid>
           </TabItem>
         </Tabs>

      3. Smart Contract erstellen

         Navigieren Sie zu Ihrem Anwendungsordner und initialisieren Sie einen neuen Smart Contract:

         ```shellscript filename="Terminal"
         aptos move init --name my_todo_list
         ```

      4. Smart Contract erstellen, kompilieren und bereitstellen!

         Das Entwickler-Setup für die Verwendung von Aptos für Smart Contracts ist nun abgeschlossen.
         Weitere Informationen finden Sie im Link zum Dapp-Tutorial unten

         <CardGrid>
           <LinkCard href="/de/build/guides/build-e2e-dapp/1-create-smart-contract#what-is-a-movetoml-file" title="Smart Contract Erstellungsleitfaden" description="Ein einfacher Todo-Listen-Leitfaden zur Einrichtung eines Smart Contracts mit Move" />
         </CardGrid>
    </Steps>
  </TabItem>

  {/* create-aptos-dapp */}

  <TabItem label="Create Aptos Dapp">
    <Steps>
      1. create-aptos-dapp installieren

         Führen Sie den folgenden Befehl aus, um eine dApp aus einer Vorlage in Sekunden zu installieren:

         <Tabs>
           { /* npx */ }

           <TabItem label="npx">
             ```shellscript filename="Terminal"
             npx create-aptos-dapp@latest
             ```
           </TabItem>

           { /* pnpx */ }

           <TabItem label="pnpx">
             ```shellscript filename="Terminal"
             pnpx create-aptos-dapp@latest
             ```
           </TabItem>
         </Tabs>

      2. Den Eingabeaufforderungen folgen

         Befolgen Sie die CLI-Eingabeaufforderungen, um einen Namen, eine [Vorlage](/de/build/create-aptos-dapp#templates) und ein Netzwerk für Ihre neue dApp auszuwählen.

         ![cad](~/images/cad-video.gif)

      3. Beginnen Sie mit der Entwicklung und Anpassung Ihrer neuen dApp!

         Navigieren Sie zu Ihrem neuen Projekt und öffnen Sie es in Ihrer bevorzugten IDE, um mit der Entwicklung fortzufahren!

         Befolgen Sie die generierte `README.md`-Datei für die nächsten Schritte.

      4. Weiterlesen

         <CardGrid>
           <LinkCard href="/de/build/create-aptos-dapp" title="Create Aptos Dapp" description="Erhalten Sie weitere Informationen über das Tool" />

           <LinkCard href="/de/build/create-aptos-dapp#templates" title="Vorlagen" description="Durchsuchen Sie vorgefertigte Vorlagen" />

           <LinkCard href="/de/build/create-aptos-dapp/faq" title="FAQ" description="Erhalten Sie Hilfe bei häufigen Problemen und Fragen" />
         </CardGrid>
    </Steps>
  </TabItem>
</Tabs>
