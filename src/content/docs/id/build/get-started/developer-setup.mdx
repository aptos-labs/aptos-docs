import { CardGrid, LinkCard, Steps, TabItem, Tabs } from '@astrojs/starlight/components';

Berikut adalah cara mudah untuk setup environment Anda tergantung pada jenis pengembangan.

<Tabs>
  {/* Frontend */}

  <TabItem label="Frontend">
    <Steps>
      1. Inisialisasi Proyek Frontend

         Berikut adalah beberapa contoh pilihan populer:

         <Tabs>
           <TabItem label="Next.js">
             ```shellscript filename="Terminal"
             pnpx create-next-app@latest
             ```
           </TabItem>

           <TabItem label="Vite (TS)">
             ```shellscript filename="Terminal"
             pnpx create vite my-aptos-app --template react-ts
             ```
           </TabItem>
         </Tabs>

      2. Install @aptos-labs/ts-sdk

         ```shellscript npm2yarn
         npm i @aptos-labs/ts-sdk
         ```

      3. Setup TS SDK

         <CardGrid>
           <LinkCard href="/id/build/sdks/ts-sdk/quickstart" title="TS SDK Quickstart" description="Lihat cara setup akun Anda, jaringan, gunakan faucet, kirim / simulasi transaksi, dan lainnya" />
         </CardGrid>

      4. Build aplikasi Anda!

         Setup developer untuk menggunakan Aptos di frontend Anda sekarang sudah selesai.
         Checkout alat-alat lain kami yang menyederhanakan proses pengembangan

         <CardGrid>
           <LinkCard href="/id/build/indexer" title="Indexer" description="Query secara efisien untuk state on-chain seperti saldo, aktivitas transaksi, data token, dan lainnya" />

           <LinkCard href="https://github.com/aptos-labs/aptos-ts-sdk/tree/main/examples/typescript" title="Contoh TS SDK" description="20+ Contoh cara menggunakan TS SDK" target="_blank" />

           <LinkCard href="https://build.aptoslabs.com/" title="Aptos Build" description="Mencapai rate limit untuk Fullnode API / Indexer? Dapatkan API Key di sini" target="_blank" />
         </CardGrid>
    </Steps>
  </TabItem>

  {/* Smart Contract */}

  <TabItem label="Kontrak Pintar">
    <Steps>
      1. Install CLI

         <CardGrid>
           <LinkCard href="/id/build/cli" title="Aptos CLI" description="Instruksi cara install Aptos CLI" />
         </CardGrid>

      2. Setup Editor atau IDE

         Tambahkan ekstensi berikut ke editor pilihan Anda untuk membuat pengembangan Move lebih mudah

         <Tabs>
           <TabItem label="JetBrains IDEs">
             <CardGrid>
               <LinkCard href="https://plugins.jetbrains.com/plugin/14721-move-language" title="Move Language Plugin" description="Plugin untuk IntelliJ IDEA, WebStorm, dan IDE JetBrains lainnya" target="_blank" />
             </CardGrid>
           </TabItem>

           <TabItem label="VSCode">
             <CardGrid>
               <LinkCard href="https://marketplace.visualstudio.com/items?itemName=AptosLabs.move-on-aptos" title="Move on Aptos Extension" description="Ekstensi resmi untuk VSCode" target="_blank" />
             </CardGrid>
           </TabItem>

           <TabItem label="Vim / Neovim">
             <CardGrid>
               <LinkCard href="https://github.com/aptos-labs/aptos-core/tree/main/third_party/move/tools/move-analyzer/editors/vim" title="Vim Plugin" description="Plugin Move untuk Vim/Neovim" target="_blank" />
             </CardGrid>
           </TabItem>
         </Tabs>

      3. Mulai dengan Template

         <CardGrid>
           <LinkCard href="/id/build/create-aptos-dapp" title="create-aptos-dapp" description="Template untuk membuat dApp Aptos dengan cepat" />

           <LinkCard href="https://github.com/aptos-labs/aptos-core/tree/main/aptos-move/move-examples" title="Contoh Move" description="30+ contoh untuk pengembangan Move di Aptos" target="_blank" />
         </CardGrid>

      4. Build kontrak pintar Anda!

         Setup developer untuk pengembangan kontrak pintar di Aptos sekarang sudah selesai.

         <CardGrid>
           <LinkCard href="/id/build/guides/first-move-module" title="Modul Move Pertama Anda" description="Tutorial step-by-step untuk membuat kontrak pintar pertama Anda" />

           <LinkCard href="/id/build/smart-contracts/book" title="Buku Move" description="Panduan komprehensif untuk bahasa pemrograman Move" />
         </CardGrid>
    </Steps>
  </TabItem>

  {/* Full-stack dApp */}

  <TabItem label="dApp Full-stack">
    <Steps>
      1. Install create-aptos-dapp

         ```shellscript filename="Terminal"
         npx create-aptos-dapp@latest
         ```

      2. Ikuti prompt untuk setup proyek Anda

         Tool ini akan membuat proyek dengan:

         - Frontend (React/Next.js/Vite)
         - Kontrak pintar Move
         - Setup wallet integration
         - Konfigurasi yang sudah dioptimalkan

      3. Mulai pengembangan!

         Setup developer untuk dApp full-stack di Aptos sekarang sudah selesai.

         <CardGrid>
           <LinkCard href="/id/build/guides/build-e2e-dapp" title="Tutorial dApp End-to-End" description="Panduan lengkap untuk membangun dApp dari awal" />
