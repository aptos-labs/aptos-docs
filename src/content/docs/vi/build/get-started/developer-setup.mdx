import { CardGrid, LinkCard, Steps, TabItem, Tabs } from '@astrojs/starlight/components';

Đây là cách dễ dàng để thiết lập môi trường của bạn tùy thuộc vào loại phát triển.

<Tabs>
  {/* Frontend */}

  <TabItem label="Frontend">
    <Steps>
      1. Khởi tạo dự án Frontend

         Đây là một số ví dụ về các lựa chọn phổ biến:

         <Tabs>
           <TabItem label="Next.js">
             ```shellscript filename="Terminal"
             pnpx create-next-app@latest
             ```
           </TabItem>

           <TabItem label="Vite (TS)">
             ```shellscript filename="Terminal"
             pnpx create vite my-aptos-app --template react-ts
             ```
           </TabItem>
         </Tabs>

      2. Cài đặt @aptos-labs/ts-sdk

         ```shellscript npm2yarn
         npm i @aptos-labs/ts-sdk
         ```

      3. Thiết lập TS SDK

         <CardGrid>
           <LinkCard
             href="/build/sdks/ts-sdk/quickstart"
             title="TS SDK Quickstart"
             description="Xem cách thiết lập tài khoản, mạng, sử dụng
         faucet, gửi / mô phỏng giao dịch và nhiều hơn nữa"
           />
         </CardGrid>

      4. Xây dựng ứng dụng của bạn!

         Việc thiết lập môi trường phát triển để sử dụng Aptos trong frontend của bạn đã hoàn tất.
         Hãy xem các công cụ khác của chúng tôi giúp đơn giản hóa quá trình phát triển

         <CardGrid>
           <LinkCard
             href="/build/indexer"
             title="Indexer"
             description="Truy vấn hiệu quả trạng thái trên chuỗi như số dư,
         hoạt động giao dịch, dữ liệu token và nhiều hơn nữa"
           />

           <LinkCard
             href="https://github.com/aptos-labs/aptos-ts-sdk/tree/main/examples/typescript"
             title="Ví dụ TS SDK"
             description="Hơn 20 ví dụ về cách sử dụng TS
         SDK"
             target="_blank"
           />

           <LinkCard
             href="https://build.aptoslabs.com/"
             title="Aptos Build"
             description="Gặp giới hạn tỷ lệ cho Fullnode API / Indexers? Lấy
         API Key tại đây"
             target="_blank"
           />
         </CardGrid>
    </Steps>
  </TabItem>

  {/* Smart Contract */}

  <TabItem label="Hợp đồng thông minh">
    <Steps>
      1. Cài đặt CLI

         <CardGrid>
           <LinkCard href="/build/cli" title="Aptos CLI" description="Hướng dẫn cách cài đặt Aptos CLI" />
         </CardGrid>

      2. Thiết lập Editor hoặc IDE

         Thêm các extension sau vào editor mà bạn chọn để làm cho việc phát triển Move
         dễ dàng hơn

         <Tabs>
           <TabItem label="JetBrains IDEs">
             <CardGrid>
               <LinkCard
                 href="https://plugins.jetbrains.com/plugin/14721-move-language"
                 title="Move on Aptos"
                 description="Language server và syntax highlighter cho
         JetBrains IDEs như CLion, Rust Rover, WebStorm,
         IntelliJ"
                 target="_blank"
               />
             </CardGrid>
           </TabItem>

           <TabItem label="VSCode">
             <CardGrid>
               <LinkCard href="https://marketplace.visualstudio.com/items?itemName=AptosLabs.move-on-aptos" title="Move on Aptos extension" description="IDE Aptos Move đầy đủ tính năng cho VSCode" target="_blank" />
             </CardGrid>
           </TabItem>

           <TabItem label="Cursor, Kiro, etc.">
             <CardGrid>
               <LinkCard href="https://open-vsx.org/extension/aptoslabs/move-on-aptos" title="Move on Aptos extension" description="IDE Aptos Move đầy đủ tính năng cho các nền tảng được hỗ trợ bởi OpenVSX" target="_blank" />
             </CardGrid>
           </TabItem>
         </Tabs>

      3. Tạo hợp đồng thông minh

         Điều hướng đến thư mục ứng dụng của bạn và khởi tạo một hợp đồng thông minh mới bằng cách
         thực hiện:

         ```shellscript filename="Terminal"
         aptos move init --name my_todo_list
         ```

      4. Xây dựng, Biên dịch và Triển khai hợp đồng thông minh!

         Việc thiết lập môi trường phát triển để sử dụng Aptos cho hợp đồng thông minh đã hoàn tất.
         Để biết thêm thông tin, hãy xem liên kết đến hướng dẫn Dapp bên dưới

         <CardGrid>
           <LinkCard
             href="/build/guides/build-e2e-dapp/1-create-smart-contract#what-is-a-movetoml-file"
             title="Hướng dẫn tạo hợp đồng thông minh"
             description="Hướng dẫn danh sách todo đơn giản về cách thiết lập một hợp đồng
         thông minh với Move"
           />
         </CardGrid>
    </Steps>
  </TabItem>

  {/* create-aptos-dapp */}

  <TabItem label="Create Aptos Dapp">
    <Steps>
      1. Cài đặt create-aptos-dapp

         Chạy lệnh dưới đây để cài đặt một dApp từ template trong vài giây:

         <Tabs>
           { /* npx */ }

           <TabItem label="npx">
             ```shellscript filename="Terminal"
             npx create-aptos-dapp@latest
             ```
           </TabItem>

           { /* pnpx */ }

           <TabItem label="pnpx">
             ```shellscript filename="Terminal"
             pnpx create-aptos-dapp@latest
             ```
           </TabItem>
         </Tabs>

      2. Theo dõi các yêu cầu

         Theo dõi các yêu cầu CLI để chọn tên, [template](/build/create-aptos-dapp#templates), và mạng cho dApp mới của bạn.

         ![cad](~/images/cad-video.gif)

      3. Bắt đầu xây dựng và tùy chỉnh dApp mới của bạn!

         Điều hướng đến dự án mới của bạn và mở trong IDE yêu thích để tiếp tục xây dựng!

         Theo dõi file `README.md` được tạo ra cho các bước tiếp theo.

      4. Tiếp tục đọc

         <CardGrid>
           <LinkCard href="/build/create-aptos-dapp" title="Create Aptos Dapp" description="Nhận thêm thông tin về công cụ này" />

           <LinkCard href="/build/create-aptos-dapp#templates" title="Templates" description="Duyệt các template được tạo sẵn" />

           <LinkCard href="/build/create-aptos-dapp/faq" title="FAQ" description="Nhận trợ giúp cho các vấn đề và câu hỏi thường gặp" />
         </CardGrid>
    </Steps>
  </TabItem>
</Tabs>
